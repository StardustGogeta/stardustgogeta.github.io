Quadratic Formula
A quadratic equation is a polynomial of degree 2.
The zeros of such a polynomial are given by the equation `(-b+-sqrt(b^2-4ac))/(2a)`.
`x^2+` `x+` 
var a = parseFloat(document.getElementById('in').value);
var b = parseFloat(document.getElementById('in2').value);
var c = parseFloat(document.getElementById('in3').value);
if(!(isNaN(a)||isNaN(b)||isNaN(c)))
{
	var x1, x2, i, r, d = b*b-4*a*c;
	if(d>=0)
	{
		x1=(-b+Math.sqrt(d))/2/a;
		x2=(-b-Math.sqrt(d))/2/a;
		return [+x1.toFixed(5),+x2.toFixed(5)];
	}
	else
	{
		r=-b/2/a;
		i=Math.sqrt(-d)/2/a;
		return [+r.toFixed(5),+i.toFixed(5)];
	}
}
