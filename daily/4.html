Radical Simplification
A radical `(\sqrt x)` symbolizes the opposite of exponentiation.
If `\sqrt[n]x=y`, then `y` can be multiplied by itself `n` times to reach `x`.
Additionally, an expression like `\sqrt{4x}` can be simplified to `2\sqrt x` because
the square root can be distributed to the factors of 4 and x, individually.

`\Large\sqrt x` 
function f(n) {
	for(var d=2;d<=Math.sqrt(n);d++) if (!(n%d)) return d;
	return n;
}
function e(a,b,c,d,e) {
	return [a*Math.pow(c,Math.floor(d/e)),b*Math.pow(c,d%e)];
}
function calc() {
	var x=parseFloat(document.getElementById('in').value);
	var n=parseFloat(document.getElementById('in2').value);
	var y=parseFloat(document.getElementById('in3').value);
	if(x&&n>0&&y>0) {
		var a=Math.pow(y,1/n);
		if(a%1==0)
			return a*x;
		else {
			var b,i=o=fV=fC=[1];
			while(y>1) {
				b=f(y);
				if(fV==b) fC+=1;
				else{a=e(o,i,fV,fC,n);fC=1;fV=b;o=a[0],i=a[1];}
				y/=b;
			}
			a=e(o,i,fV,fC,n);
			return [a[0]*x,n,a[1]];
		}
	}
}
